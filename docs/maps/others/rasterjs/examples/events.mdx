# Обработка событий

## Описание

Ниже приведены примеры работы с событиями. Для получения подробной информации см. разделы [DG.Evented](/maps/others/rasterjs/reference/base-classes#dgevented) и [Events](/maps/others/rasterjs/reference/base-classes#events).

## Подписка на события

Пример подписки на различные события (клик в маркер, карту, геометрию):

<HtmlExampleWithSrcCode iframeStyles={{ height: '300px' }}>
{`
    <html>
        <head>
            <title>RasterJS API</title>
            <meta name="description" content="Подписка на события" />
            <script src="https://maps.api.2gis.ru/2.0/loader.js"></script>
        </head>
        <body>
            Вы кликнули на: <span id="clicked_element"></span>
            <div id="map" style="width: 100%; height: 400px;"></div>
            <script>
                DG.then(function() {
                    var map,
                        clickedElement = document.getElementById('clicked_element'),
                        coords = [
                            [54.99, 82.88],
                            [54.985, 82.94],
                            [54.984, 82.925],
                            [54.981, 82.928]
                        ];

                    map = DG.map('map', {
                        center: [54.98, 82.89],
                        zoom: 13,
                        key: 'Your API access key'
                    });

                    map.on('click', function(e) {
                        clickedElement.innerHTML = 'карту, координаты ' + e.latlng.lat + ', ' + e.latlng.lng;
                    });

                    DG.marker([54.98, 82.89])
                        .on('click', function() {
                            clickedElement.innerHTML = 'маркер';
                        })
                        .addTo(map);

                    DG.polygon(coords)
                        .on('click', function() {
                            clickedElement.innerHTML = 'многоугольник';
                        })
                        .addTo(map);
                });
            </script>
        </body>
    </html>
`}
</HtmlExampleWithSrcCode>

## Подписка на изменение проекта <Brand />

При изменении [проекта](/maps/others/rasterjs/reference/dg-project-detector) выделяются границы текущего:

<HtmlExampleWithSrcCode iframeStyles={{ height: '300px' }}>
{`
<html>
    <head>
        <title>RasterJS API</title>
        <meta name="description" content="Подписка на изменение проекта" />
        <script src="https://maps.api.2gis.ru/2.0/loader.js"></script>
    </head>
    <body>
        <div id="map" style="width: 100%; height: 400px"></div>
        <script>
            DG.then(function () {
                var map, currentProjectBound;

                map = DG.map('map', {
                    center: DG.latLng(54.98, 82.89),
                    zoom: 9,
                    key: 'Your API access key'
                });

                // Подписка на событие изменения текущего проекта
                map.on('projectchange', function (e) {
                    var bounds = e.getProject().bound;
                    currentProjectBound = DG.geoJSON(bounds, {
                        color: '#f03',
                        weight: 1,
                    }).addTo(map);
                });

                // Подписка на событие выхода из проекта
                map.on('projectleave', function (e) {
                    if (currentProjectBound) {
                        map.removeLayer(currentProjectBound);
                    }
                });
            });
        </script>
    </body>
</html>
`}
</HtmlExampleWithSrcCode>
