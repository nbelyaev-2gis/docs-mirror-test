# Векторные слои

## Описание

Ниже приведены базовые примеры использования геометрических объектов. Для получения подробной информации о работе с векторными слоями см. раздел [Векторные слои](/maps/others/rasterjs/reference/vector-layers).

## Отображение прямоугольников

<HtmlExampleWithSrcCode iframeStyles={{ height: '300px' }}>
{`
<html>
    <head>
        <title>RasterJS API</title>
        <meta name="description" content="Отображение прямоугольников" />
        <script src="https://maps.api.2gis.ru/2.0/loader.js"></script>
    </head>
    <body>
        <div id="map" style="width: 100%; height: 400px"></div>
        <script>
            DG.then(function () {
                var map,
                    coords1 = [
                        [54.98, 82.87],
                        [54.975, 82.91],
                    ],
                    coords2 = [
                        [54.985, 82.87],
                        [54.982, 82.91],
                    ],
                    coords3 = [
                        [54.99, 82.87],
                        [54.987, 82.91],
                    ],
                    rectangles = DG.featureGroup();

                map = DG.map('map', {
                    center: [54.98, 82.89],
                    zoom: 14,
                    key: 'Your API access key'
                });

                // Добавление прямоугольников в группу
                DG.rectangle(coords1, { color: 'blue' }).addTo(rectangles);
                DG.rectangle(coords2, { color: 'red' }).addTo(rectangles);
                DG.rectangle(coords3, { color: 'green' }).addTo(rectangles);

                // Добавление группы на карту
                rectangles.addTo(map);

                // Подстройка границ карты
                map.fitBounds(rectangles.getBounds());
            });
        </script>
    </body>
</html>
`}
</HtmlExampleWithSrcCode>

## Векторные слои с подсказками и попапами

<HtmlExampleWithSrcCode iframeStyles={{ height: '300px' }}>
{`
<html>
    <head>
        <title>RasterJS API</title>
        <meta name="description" content="Векторные слои с подсказками и попапами" />
        <script src="https://maps.api.2gis.ru/2.0/loader.js"></script>
    </head>
    <body>
        <div id="map" style="width: 100%; height: 400px"></div>
        <script>
            DG.then(function () {
                var map = DG.map('map', {
                    center: [54.98, 82.89],
                    zoom: 14,
                    key: 'Your API access key'
                });

                DG.circle([54.98, 82.87], 200, { color: 'red' })
                    .bindPopup('Круг')
                    .bindLabel('Нажмите на круг')
                    .addTo(map);

                DG.circleMarker([54.985, 82.89])
                    .bindPopup('Круглый маркер')
                    .bindLabel('Нажмите на круглый маркер')
                    .setRadius(100)
                    .addTo(map);

                DG.rectangle(
                    [
                        [54.98, 82.87],
                        [54.975, 82.91],
                    ],
                    { color: 'green' },
                )
                    .bindPopup('Прямоугольник')
                    .bindLabel('Нажми на прямоугольник')
                    .addTo(map);

                DG.polygon(
                    [
                        [54.985, 82.875],
                        [54.98, 82.96],
                        [54.979, 82.92],
                    ],
                    { color: 'yellow' },
                )
                    .bindPopup('Многоугольник')
                    .bindLabel('Нажмите на многоугольник')
                    .addTo(map);
            });
        </script>
    </body>
</html>
`}
</HtmlExampleWithSrcCode>

## Анимация отрисовки ломаной

<HtmlExampleWithSrcCode iframeStyles={{ height: '300px' }}>
{`
<html>
    <head>
        <title>RasterJS API</title>
        <meta name="description" content="Анимация отрисовки ломаной" />
        <script src="https://maps.api.2gis.ru/2.0/loader.js"></script>
    </head>
    <body>
        <input id="playAnimation" type="button" value="Запустить анимацию" />
        <div id="map" style="width: 100%; height: 400px"></div>
        <script>
            var playAnimationButton = document.getElementById('playAnimation');

            DG.then(function () {
                var map = DG.map('map', {
                    center: [54.98, 82.94],
                    zoom: 13,
                    key: 'Your API access key'
                });

                playAnimation.onclick = function () {
                    var polyline = DG.polyline([]).addTo(map),
                        counter = 0;

                    (function draw() {
                        polyline.addLatLng([54.98, 82.89 + counter / 10000]);

                        if (++counter < 1000) {
                            window.setTimeout(draw, 10);
                        } else {
                            counter = 0;
                            draw();
                        }
                    })();
                };
            });
        </script>
    </body>
</html>
`}
</HtmlExampleWithSrcCode>
